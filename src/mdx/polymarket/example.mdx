
## 哪里查询
https://dune.com/queries/5307268/8706778

```sql
WITH
  orders_filled AS (
    SELECT
      maker,
      taker,
      evt_block_time
    FROM
      polymarket_polygon.CTFExchange_evt_OrderFilled
    WHERE
      evt_block_time > CURRENT_TIMESTAMP - INTERVAL '3' MONTH
  )
SELECT
  maker,
  taker,
  COUNT(*) AS num_orders,
  DATE_TRUNC('day', evt_block_time) AS day
FROM
  orders_filled
WHERE
  (CAST(maker AS VARCHAR) LIKE '0x80cd%' AND CAST(maker AS VARCHAR) LIKE '%0273')
  OR (CAST(taker AS VARCHAR) LIKE '0x80cd%' AND CAST(taker AS VARCHAR) LIKE '%0273')
GROUP BY
  maker,
  taker,
  DATE_TRUNC('day', evt_block_time)
ORDER BY
  day DESC,
  num_orders DESC
LIMIT 100;

```
